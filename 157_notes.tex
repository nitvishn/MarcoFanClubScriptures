\section{Fundamental Theorems of Calculus}

\subsection{FTC1}

\[F(x) = \int_a^x f \implies F'(x_0) = f(x_0)\]

\begin{itemize}
    \item Preconditions:
    \begin{itemize}
        \item $f: [a, b] \to \R$ is \textbf{bounded}, \textbf{integrable}, and \textbf{continuous} at $x_0 \in (a, b)$
        \item $F: [a, b] \to \R$ is defined as $\int_a^x f$ 
    \end{itemize}
    \item Implications:
    \begin{itemize}
        \item $F$ is differentiable at $x_0 \in (a, b)$
        \item $F'(x_0) = f(x_0)$
        \item Loosely: 
    \end{itemize}
\end{itemize}

\subsection{FTC2}


\[\int_a^b f = F(b) - F(a)\]

\begin{itemize}
    \item Preconditions:
    \begin{itemize}
        \item $f: [a, b] \to \R$ is \textbf{bounded} and \textbf{integrable}
        \item $F: [a, b] \to \R$ is \textbf{continuous} and \textbf{differentiable} on $(a, b)$
        \item $F' = f$ on $(a, b)$
    \end{itemize}
    \item Implications:
    \begin{itemize}
        \item Stated above. 
    \end{itemize}
    \item Remarks:
    \begin{itemize}
        \item If $f$ is continuous, we can define $F = \int_a^x f$. Then from FTC1, $F'$ is differentiable. 
    \end{itemize}
    \item Warnins: 
    \begin{itemize}
        \item We need that $f = F'$ is integrable. If we have $F$ continuous and differentiable and $F'$ bounded, then $F'$ integrable \textbf{does not necessarily follow}. See \href{https://en.wikipedia.org/wiki/Volterra%27s_function}{Volterra's function}. 
    \end{itemize}  
\end{itemize}

\section{Integration}

\begin{itemize}
    \item The integral \[\int_0^1 t^\lambda dt\] converges \textbf{if and only if} $\lambda > -1 $. $\lambda = -1$ is the borderline case. 
    \item The integral \[\int_1^\infty t^\lambda dt\] converges \textbf{if and only if} $\lambda < - 1$.
\end{itemize}
